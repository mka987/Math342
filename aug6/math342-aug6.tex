\noindent{\bf Question 1.} Find the rational number, expressed in lowest terms,
represented by each of the following simple continued fractions.
\begin{enumerate*}[{\bf (a)}]
\item $[10; 3]$,
\item $[3; 2, 1]$,
\item $[0; 1, 2, 3]$,
\item $[2; 1, 2, 1]$,
\item $[2; 1, 2, 1, 1, 4]$,
\item $[1; 2, 1, 2]$,
\item $[1; 2, 1, 2, 1]$,
\item $[1; 2 1, 2, 1, 2]$.
\end{enumerate*} \\

\noindent{\bf Question 2.} Find the simple continued fraction expansion, not
terminating with the partial quotient of 1, of each of the following rational
numbers.
\begin{enumerate*}[{\bf (a)}]
\item $18/13$,
\item $32/17$,
\item $19/9$,
\item $310/99$,
\item $-931/1005$,
\item $831/8110$.
\end{enumerate*} \\

\noindent{\bf Question 3.} Let $f_k$ denote the $k$th Fibonacci number. Find the
simple continued fraction, terminating with the partial quotient $f_{k+1}/f_k$,
where $k$ is a positive integer. \\

\noindent{\bf Question 4.} Let $[a_0; a_1,\dots,a_n]$ be the simple continued
fraction expansion of $r/s$, where $(r,s)=1$ and $r \geqq 1$. Show that this
continued fraction is symmetric, that is, $a_0=a_n$, $a_1=a_{n-1}$, $\dots$, if
and only if $r \mid (s^2+1)$ if $n$ is odd and $r \mid (s^2-1)$ if $n$ is even.
